Kotlin is like Java
BASICS
Hello World
Java
                            System.out.printl("Hello, world!");
Kotlin
                            println("Hello, world!")
Variables And Constants
Java
                            int myVariable = 42;
myVariable = 50;
final myConstant = 42;
Kotlin
                            var myVariable = 42
myVariable = 50
val myConstant = 42
Explicit Types
Java
                            double explicitDouble = 70.0;
Kotlin
                            var explicitDouble: Double = 70.0
String Interpolation
Java
                            int apples = 3;
int oranges = 5;
String fruitSummary = "I have "
+ (apples + oranges) + " pieces of fruit";
Kotlin
                            var apples = 3
var oranges = 5
var fruitSummary = "I have ${apples + oranges}
 pieces of fruit."
Converting Types
Java
                            float piecesOfBiscuits = 9.6f;
int biscuits = Integer.parseInt(piecesOfBiscuits);
Kotlin
                            var piecesOfBiscuits = 9.6f
var biscuits = piecesOfBiscuits.toInt()
Range Operator
Java
                            final String names[] = {"Anna", "Alex", "Brian", "Jack"};
final int count = names.length;

for (int i = 0; i < count-1; i++) {
    System.out.println(String.format("Person %s is called %s",
    count + 1, names[i]));
}
// Person 1 is called Anna
// Person 2 is called Alex
// Person 3 is called Brian
// Person 4 is called Jack
Kotlin
                            val names = arrayOf("Anna", "Alex", "Brian", "Jack")
val count = names.count()

for (i in 0..count - 1) {
    println("Person ${i + 1} is called ${names[i]}")
}
// Person 1 is called Anna
// Person 2 is called Alex
// Person 3 is called Brian
// Person 4 is called Jack
